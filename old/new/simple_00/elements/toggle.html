TOGGLE

DEMO 

<button class="toggle">toggle button</button>
	

.toggle {
  background: red;
}

.toggle.active {
  background: blue;
}
	

toggle();
				
function toggle() {
  document.querySelectorAll('.toggle').forEach((item) =>
    item.addEventListener('click', () => {
    item.classList.toggle('active');
    })
  )
}
