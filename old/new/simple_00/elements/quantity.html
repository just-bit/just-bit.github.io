<div class="quantity">
  <button class="quantity__btn minus-btn disabled" type="button">Минус</button>
  <input class="quantity__value" id="quantity" value="1" type="text">
  <button class="quantity__btn plus-btn" type="button">Плюс</button>
  <p class="quantity__price">
    <span class="quantity__price-num" id="price">3</span>
  </p>
</div>
	

quantity();

function quantity() {
document.querySelector(".minus-btn").setAttribute("disabled", "disabled");
let valueCount;
let price = document.getElementById("price").innerText;
function priceTotal() {
let total = valueCount * price;
document.getElementById("price").innerText = total
}
document.querySelector(".plus-btn").addEventListener("click", function () {
valueCount = document.getElementById("quantity").value;
valueCount++;
document.getElementById("quantity").value = valueCount;
if (valueCount > 1) {
document.querySelector(".minus-btn").removeAttribute("disabled");
document.querySelector(".minus-btn").classList.remove("disabled")
}
priceTotal()
})
document.querySelector(".minus-btn").addEventListener("click", function () {
valueCount = document.getElementById("quantity").value;
valueCount--;
document.getElementById("quantity").value = valueCount;
if (valueCount == 1) {
document.querySelector(".minus-btn").setAttribute("disabled", "disabled")
}
priceTotal()
})
}
