function starRating() {
  let ratingItem = document.querySelectorAll('.rating__item');
  let ratingNum = document.querySelector('.num');

  ratingItem.forEach(item =>
    item.addEventListener('click', () => {
      item.parentNode.dataset.totalValue = item.dataset.itemValue;
      ratingNum.innerHTML = item.dataset.itemValue;
    }
  ));
}
