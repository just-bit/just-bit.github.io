SWIPER

Еслм слайдер находится внутри flex-элемента, то этому flex-элементу пишем min-width: 0;
Если слайдер находится внутри изначально скрытого элемента (напр. табы) ------------
Чтобы в "pagination" выводить несколько элементов навигации ----------------- 


<link rel="stylesheet" href="swiper-bundle.min.css">

<script src="swiper-bundle.min.js"></script>

<div class="my-slider swiper-container">
  <div class="swiper-wrapper">
    <div class="swiper-slide">Slide 1</div>
    <div class="swiper-slide">Slide 2</div>
    <div class="swiper-slide">Slide 3</div>
  </div>

  <div class="swiper-button-prev"></div>
  <div class="swiper-button-next"></div>

  <div class="swiper-pagination"></div>
  <div class="swiper-scrollbar"></div>
</div>
	

let myImageSlider = new Swiper('.my-slider', {

// Стрелки ==>
navigation: {
  nextEl: '.swiper-button-next',
  prevEl: '.swiper-button-prev'
  },

// Навигация ==>
pagination: {
  el: '.swiper-pagination',
  // - Буллеты (один из типов)
  type: 'bullets',
  clickable: true,
  // -- Динамические буллеты
  dynamicBullets: true,
  // -- Кастомные буллеты (номер внутри булета)
  renderBullet: function (index, className) {
    return '' + (index + 1) + '';
  },
  // - Фракция (один из типов)
  type: 'fraction',
  // -- Кастомный вывод фракции (Фото 2 из 7)
  renderFraction: function (currentClass, totalClass) {
    return 'Фото <span class="' + currentClass + '"></span>' + ' из ' + '<span class="' + totalClass + '"></span>';
  },
  // - Прогрессбар (один из типов)
  type: 'progressbar'
},

// Скролл ==>
scrollbar: {
  el: '.swiper-scrollbar',
  // - Возможность перетаскивать скролл
  draggable: true
},

// Отступ между слайдами
spaceBetween: 30,

// Автовысота слайдера
autoHeight: false,

// Количество пролистываемых слайдов за раз
slidesPerGroup: 1,
	
// Активный слайд по центру (true для активации)
  centeredSlides: false,
	
// Стартовый слайд
initialSlide: 0,

// Количество слайдов для показа (можно 'auto', 3.5, 3.6 ...)
// если slidesPerView: 'auto' ==> .swiper-slide {width: auto;}
slidesPerView: 3,
	
// Отключение функционала
// если слайдов меньше чем нужно (slidesPerView)
watchOverflow: true,

// Перетаскивание на ПК
simulateTouch: true,

// Курсор перетаскивания
grabCursor: true,

// Чувствительность свайпа (0 - выкл. свайпа на всех устройствах)
// 1< требуется меньше усилий
touchRatio: 1,

// Угол срабатывания свайпа/перетаскивания
touchAngle: 45,

// Переключение при клике на слайд
slideToClickedSlide: false,

// Навигация по хешу ==>
// для каждого .swiper-slide пишем свой data-hash="myslide-my$"
hashNavigation: {
  // - Навигация с помощью стрелок браузера
  watchState: true,
},

// Управление клавиатурой ==>
keyboard: {
enabled: true,
// - только когда слайдер в пределах вьюпорта
onlyInViewport: true,
// - управление клавишами pageUp, pageDown
pageUpDown: true,
},

// Управление колесом мыши ==>
mousewheel: {
  // - Чувствительность колеса мыши
  sensitivity: 1,
  // - Класс объекта на котором, будет срабатывать прокрутка мышью.
  // - Если слайдеров несколько eventsTarget можно не писать
  eventsTarget: '.my-slider'
},


});
